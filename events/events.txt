country_event = {
	id = KOR.9
	title = KOR.9.t
	desc = KOR.9.desc
	picture = GFX_report_event_ger_civil_war

	fire_only_once = yes

	is_triggered_only = yes

	
	immediate = {
		hidden_effect = {
			add_stability = 30
			every_state = {
				limit = {
					is_owned_by = KOR
					NOT = { has_state_flag = KOR_has_blackshirt_support }
				}
				set_state_flag = KOR_has_no_blackshirt_support
			}
			random_owned_state = {
				limit = {
					has_state_flag = KOR_has_blackshirt_support
				}
				KOR = { set_capital = PREV }
			}
			start_civil_war = {
				ruling_party = democratic
				ideology = fascism
				size = 0.5
				states = all
				states_filter = {
					has_state_flag = KOR_has_no_blackshirt_support
				}
			}
			set_politics = {
				ruling_party = democratic
				elections_allowed = no
			}
			add_popularity = {
				ideology = democratic
				popularity = 0.51
			}
			add_popularity = {
				ideology = communism
				popularity = 0.05
			}
			add_popularity = {
				ideology = fascism
				popularity = 0.15
			}
			random_other_country = {
				limit = {
					original_tag = KOR
					has_government = fascism
				}
				save_event_target_as = democratic_uk_civil_war
				every_unit_leader = {
					limit = {
						OR = {
							has_id = 999
						}
					}
					set_nationality = KOR
				}
							set_nationality = event_target:democratic_uk_civil_war
			}
			event_target:democratic_uk_civil_war = {
				if = {
					limit = {
						has_full_control_of_state = 797
					}
					set_capital = 797
				}
				else_if = {
					limit = {
						has_full_control_of_state = 798
					}
					set_capital = 798
				}
				else_if = {
					limit = {
						has_full_control_of_state = 801
					}
					set_capital = 801
				}
				else = {
					random_owned_state = {
						limit = {
							is_on_continent = asian
						}
						event_target:democratic_uk_civil_war = { set_capital = PREV }
					}
				}
			}
			event_target:democratic_uk_civil_war = {
				every_owned_state = {
					limit = {
						is_on_continent = asian
					}
				}
			}
		}
	}
	
	# Ok
	option = {
		name = KOR.9.a
		custom_effect_tooltip = ENG_democratic_civil_war_tt
		hidden_effect = {
			RAJ = { retire_country_leader = yes }
		}
		hidden_effect = { news_event = { id = mtg_news.41 days = 1 } }
	}
}
